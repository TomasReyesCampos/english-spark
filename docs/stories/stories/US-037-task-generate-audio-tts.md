## Contexto

Generar audio TTS para los ~240 audio atomics del seed (incluyendo
240+ character-bound audios + generic phrases). ElevenLabs es el
proveedor primario por calidad para conversational register.

Esta task es consumida por **EPIC-02 Content pipeline** (atomics
generation batch). No es invocada en runtime del producto — se
ejecuta solo durante creación de contenido.

## Scope

### In

- Implementar **ElevenLabsAdapter** completo:
  - `invokeAudio` con ElevenLabs API (multilingual_v2 model).
  - Voice settings: stability 0.5, similarity_boost 0.75,
    style 0.4, use_speaker_boost true.
  - Cost estimation desde character count (free tier 10k/month
    luego $0.02 per 1k chars).
- Registrar task `generate_audio_tts`:
  - Provider chain: `[{elevenlabs, eleven_multilingual_v2, weight: 100}]`.
  - Budget: $0.005 per call (típico atomic ~25 chars = $0.0005).
  - Timeout: 30s.
  - Active: true.
- Input schema:
  ```typescript
  {
    text: string,
    voice_id: string,                // de voice pool seed §3
    speed?: number,                  // 0.85-1.15
    output_format?: 'mp3_22050_32',  // default
  }
  ```
- Output schema:
  ```typescript
  {
    audio_url: string,               // R2 URL del file uploaded
    storage_key: string,
    duration_ms: number,
    character_count: number,
    cost_usd: number,
  }
  ```
- Post-process pipeline:
  - Recibir audio MP3 de ElevenLabs.
  - Trim silence at ends (FFmpeg).
  - Normalize -16 LUFS (loudness).
  - Upload a R2 con `storage_key` provisto en input (o
    autogenerated).
- Telemetry: `tts.generated`,
  `tts.normalization_applied`.

### Out

- Voice cloning pipeline (post-MVP — para Anita host voice).
- Real-time TTS streaming (post-MVP).
- Multiple language TTS (solo EN en MVP; ES post-MVP cuando
  integremos).

## Acceptance criteria

- **Given** task registrada y activa, **When** se invoca con
  `text="Hello, how are you?", voice_id="voice_us_f_25_warm"`,
  **Then** retorna 200 con `audio_url` accesible y
  `duration_ms ≈ 2000`.
- **Given** voice_id no en pool válido, **When** ElevenLabs rechaza,
  **Then** retorna error `invalid_request`.
- **Given** text > 5000 chars (límite ElevenLabs single call),
  **When** task valida, **Then** retorna 400 antes de invocar
  ElevenLabs.
- **Given** audio recibido tiene 1.5s silence al final, **When**
  post-process trim, **Then** audio final no tiene esa silence.
- **Given** audio crudo a -10 LUFS, **When** normalization aplica,
  **Then** audio final está a -16 LUFS ± 1 (verificable via
  ffprobe).
- **Given** ElevenLabs rate-limited, **When** falla con 429,
  **Then** retry 1x con backoff. Sin fallback en MVP (single
  provider).
- **Given** R2 upload falla, **When** post-process termina,
  **Then** retorna error `storage_failed` y se loguea SEV-2.
- **Given** invocation con 25 chars, **When** cost se calcula,
  **Then** `cost_usd ≈ 0.0005` (verificable).

```typescript
// apps/workers/ai-gateway/src/adapters/elevenlabs.ts
export class ElevenLabsAdapter implements ProviderAdapter {
  readonly providerId = 'elevenlabs';
  readonly supportedModels = ['eleven_multilingual_v2'];

  constructor(private env: Env) {}

  async invokeAudio(
    input: { text: string; voice_id: string; speed?: number },
    config: ProviderConfig
  ): Promise<{ audio_buffer: ArrayBuffer; character_count: number; latency_ms: number }> {
    const start = Date.now();

    if (input.text.length > 5000) {
      throw new ProviderError('invalid_request', 'Text exceeds 5000 char limit');
    }

    const response = await fetch(
      `https://api.elevenlabs.io/v1/text-to-speech/${input.voice_id}`,
      {
        method: 'POST',
        headers: {
          'Accept': 'audio/mpeg',
          'Content-Type': 'application/json',
          'xi-api-key': this.env.ELEVENLABS_API_KEY,
        },
        body: JSON.stringify({
          text: input.text,
          model_id: config.model,
          voice_settings: {
            stability: 0.5,
            similarity_boost: 0.75,
            style: 0.4,
            use_speaker_boost: true,
          },
        }),
      }
    );

    if (response.status === 429) {
      throw new ProviderError('rate_limited', 'ElevenLabs rate limit');
    }
    if (response.status === 422) {
      throw new ProviderError('invalid_request', 'Invalid voice_id or params');
    }
    if (!response.ok) {
      throw new ProviderError('provider_unavailable', `Status ${response.status}`);
    }

    const audioBuffer = await response.arrayBuffer();
    return {
      audio_buffer: audioBuffer,
      character_count: input.text.length,
      latency_ms: Date.now() - start,
    };
  }

  estimateCost(input: { text: string }): number {
    return (input.text.length / 1000) * 0.02;
  }

  async isHealthy(): Promise<boolean> {
    try {
      const response = await fetch('https://api.elevenlabs.io/v1/voices', {
        headers: { 'xi-api-key': this.env.ELEVENLABS_API_KEY },
      });
      return response.ok;
    } catch { return false; }
  }
}

// apps/workers/ai-gateway/src/handlers/generate-audio-tts.ts
export async function handleGenerateAudioTTS(input: any, context: HandlerContext) {
  const validated = InputSchema.parse(input);

  const result = await invokeWithFallback('generate_audio_tts', validated, context);
  const { audio_buffer, character_count } = result.output;

  // Post-process: trim + normalize via FFmpeg
  const processed = await processAudio(audio_buffer);

  // Upload to R2
  const storageKey = validated.storage_key
    ?? `atomics/v1/${crypto.randomUUID()}.mp3`;
  await context.env.R2_BUCKET.put(storageKey, processed.buffer, {
    httpMetadata: { contentType: 'audio/mpeg' },
  });

  track('tts.generated', {
    voice_id: validated.voice_id,
    character_count,
    duration_ms: processed.duration_ms,
  });

  return {
    audio_url: `https://r2.englishspark.com/${storageKey}`,
    storage_key: storageKey,
    duration_ms: processed.duration_ms,
    character_count,
    cost_usd: result.metadata.cost_usd,
  };
}

async function processAudio(buffer: ArrayBuffer): Promise<{ buffer: ArrayBuffer; duration_ms: number }> {
  // FFmpeg WASM trim silence + normalize -16 LUFS
  // (Detalle de implementación dependiente del setup de FFmpeg en Workers)
  // Pseudo:
  const trimmed = await ffmpeg.trim_silence(buffer);
  const normalized = await ffmpeg.normalize_lufs(trimmed, -16);
  const duration = await ffmpeg.duration(normalized);
  track('tts.normalization_applied');
  return { buffer: normalized, duration_ms: duration };
}
```
